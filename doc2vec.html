from gensim.models.doc2vec import Doc2Vec, TaggedDocument

list_text = list(df["text"])
list_title = list(df["title"])

tagged_data = [TaggedDocument(words=word_tokenize(term_def.lower()), tags=[list_title[i]]) for i, term_def in enumerate(list_text)]

max_epochs = 100
vec_size = 100
alpha = 0.025

model = Doc2Vec(vector_size=vec_size,
                alpha=alpha, 
                min_alpha=0.00025,
                min_count=1,
                dm=1)
  
model.build_vocab(tagged_data)
